# Extension secrets for s3filestore
# Format: ENV_VAR_NAME=secret-name:json-key
# 
# The secret-name should match a secret in AWS Secrets Manager.
# The json-key is the key within the JSON secret to extract.
#
# Available variables for substitution:
#   ${SECRET_NAME}    - Main project secret: ${UNIQUE_PROJECT_ID}-${ENVIRONMENT}-secrets
#   ${UNIQUE_PROJECT_ID} - Project identifier from .env
#   ${ENVIRONMENT}    - Environment name (dev, prod, etc.)
#   ${AWS_REGION}     - AWS region
#
# These variables will be exported before extension.entrypoint.sh runs.

S3FILESTORE_AWS_BUCKET_NAME=${SECRET_NAME}:s3filestore_aws_bucket_name
S3FILESTORE_REGION_NAME=${SECRET_NAME}:s3filestore_region_name
S3FILESTORE_AWS_ACCESS_KEY_ID=${SECRET_NAME}:s3filestore_aws_access_key_id
S3FILESTORE_AWS_SECRET_ACCESS_KEY=${SECRET_NAME}:s3filestore_aws_secret_access_key
